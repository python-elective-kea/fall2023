Ex 6: Extract .py files

 1import sys
 2import os
 3import subprocess
 4from zipfile import ZipFile 
 5
 6def main(argv):
 7
 8    input_check(argv)
 9
10    # A
11    # list all files in folder
12    x = os.listdir(argv[1])
13    
14    # list all .py files
15    pyfiles = [i for i in x if i[-3:] == '.py']
16    
17    # create new dir
18    os.mkdir(argv[3])
19
20    # copy all files in list --todir
21    for i in pyfiles:
22        subprocess.run(['cp', i, argv[3]])
23
24    #B
25    with ZipFile(argv[3]+'/archive.zip', 'w') as zf:        
26            for file in pyfiles:
27                zf.write(file)
28
29def input_check(argv):
30    if len(argv) < 2:
31        print('Usage: python extract.py [.] [--todir <<tmp/>>] {--zip <<filename.zip>>}')
32        sys.exit()
33    if len(argv) == 3 and argv[2] != '--todir':
34        print('Usage: python extract.py [.] [--todir <<tmp/>>] {--zip <<filename.zip>>}')
35        sys.exit()

37main(sys.argv)
